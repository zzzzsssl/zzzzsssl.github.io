#UnityTips: ✨ Capsule/Line | #Unity3D 💊

float W = 0.2, L = 0.5;

float W2=W*L/2.,L2=L-(W2*2.),h,lineSDF,line;
vec2 uv=(FC.xy-.5*r)/r.y,s=vec2(0,-L2/2.),e=vec2(0,L2/2.),p1=uv-s,p2=e-s;

h=clamp(dot(p1,p2)/dot(p2,p2),0.,1.);

lineSDF=length(p1-p2*h)-W2;
line=step(lineSDF,0.); pic.twitter.com/5Tf19QzrIA
